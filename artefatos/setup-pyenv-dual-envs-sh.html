  9Options:

 16  --skip-rnc                Nao processa ambiente legado RNC

 19  -h, --help                Mostra esta ajuda

 21Examples:
 22  scripts/dev/setup_pyenv_dual_envs.sh --dry-run
 23  scripts/dev/setup_pyenv_dual_envs.sh --set-local

 25EOF
 26}

 50pyenv_has_version() {
 51    local version="$1"
 52    local line
 53    local found=1
 54    while IFS= read -r line; do
 55        [[ "$line" == "$version" ]] && found=0
 56    done < <(pyenv versions --bare)
 57    return "$found"
 58}
 59
 60pyenv_has_virtualenv() {
 61    local env_name="$1"
 62    local line
 63    local found=1
 64    while IFS= read -r line; do
 65        [[ "$line" == "$env_name" ]] && found=0
 66    done < <(pyenv virtualenvs --bare)
 67    return "$found"
 68}

 70install_python_if_missing() {
 71    local version="$1"
 72    if pyenv_has_version "$version"; then
 73        log "Python $version ja instalado."
 74        return 0
 75    fi
 76    log "Instalando Python $version via pyenv..."
 77    run_cmd pyenv install "$version"
 78}
 79
 80create_virtualenv_if_missing() {
 81    local version="$1"
 82    local env_name="$2"
 83    if pyenv_has_virtualenv "$env_name"; then
 84        log "Virtualenv '$env_name' ja existe."
 85        return 0
 86    fi
 87    log "Criando virtualenv '$env_name' (Python $version)..."
 88    run_cmd pyenv virtualenv "$version" "$env_name"
 89}

 91install_requirements() {
 92    local env_name="$1"
 93    local req_file="$2"
 94    [[ -f "$req_file" ]] || die "arquivo de requirements nao encontrado: $req_file"
 95
 96    log "Atualizando pip/setuptools/wheel no ambiente '$env_name'..."
 97    run_cmd env PYENV_VERSION="$env_name" python -m pip install --upgrade pip setuptools wheel
 98
 99    log "Instalando dependencias de '$req_file' no ambiente '$env_name'..."
100    run_cmd env PYENV_VERSION="$env_name" python -m pip install -r "$req_file"
101}
