  1#!/usr/bin/env bash
  2set -euo pipefail
  3
  4usage() {
  5    cat <<'EOF'
  6Usage:
  7  scripts/dev/run_fluxo_isolated.sh [options] -- <args do fluxo_sismo.sh>
  8
  9Options:
 10  --ref <git-ref>         Commit/branch/tag usado no worktree (default: HEAD)
 11  --worktree-dir <path>   Diretorio do worktree (default: mktemp em /tmp)
 12  --keep-worktree         Nao remove o worktree ao final
 13  -h, --help              Mostra esta ajuda
 14
 15Examples:
 16  scripts/dev/run_fluxo_isolated.sh -- catalogo_jul.csv -e -t
 17  scripts/dev/run_fluxo_isolated.sh --ref HEAD~1 -- catalogo_jul.csv -pe
 18EOF
 19}
 20
 21SCRIPT_DIR="$(cd "$(dirname "${BASH_SOURCE[0]}")" && pwd)"
 22REPO_ROOT="$(git -C "$SCRIPT_DIR" rev-parse --show-toplevel 2>/dev/null || true)"
 23
 24if [[ -z "$REPO_ROOT" ]]; then
 25    echo "Erro: nao foi possivel localizar a raiz do repositorio via git." >&2
 26    exit 1
 27fi
 28
 29REF="HEAD"
 30KEEP_WORKTREE=0
 31WORKTREE_DIR=""
 32WORKTREE_ADDED=0
 33
 34while [[ $# -gt 0 ]]; do
 35    case "$1" in
 36        --ref)
 37            REF="${2:-}"
 38            if [[ -z "$REF" ]]; then
 39                echo "Erro: --ref requer um valor." >&2
 40                exit 1
 41            fi
 42            shift 2
 43            ;;
 44        --worktree-dir)
 45            WORKTREE_DIR="${2:-}"
 46            if [[ -z "$WORKTREE_DIR" ]]; then
 47                echo "Erro: --worktree-dir requer um caminho." >&2
 48                exit 1
 49            fi
 50            shift 2
 51            ;;
 52        --keep-worktree)
 53            KEEP_WORKTREE=1
 54            shift
 55            ;;
 56        -h|--help)
 57            usage
 58            exit 0
 59            ;;
 60        --)
 61            shift
 62            break
 63            ;;
 64        *)
 65            break
 66            ;;
 67    esac
 68done
 69
 70if [[ $# -eq 0 ]]; then
 71    echo "Erro: informe os argumentos do fluxo_sismo.sh apos '--'." >&2
 72    usage
 73    exit 1
 74fi
 75
 76if [[ -z "$WORKTREE_DIR" ]]; then
 77    WORKTREE_DIR="$(mktemp -d -t classificador-fluxo-XXXXXX)"
 78else
 79    if [[ -e "$WORKTREE_DIR" ]]; then
 80        echo "Erro: diretorio '$WORKTREE_DIR' ja existe. Use outro caminho." >&2
 81        exit 1
 82    fi
 83    mkdir -p "$WORKTREE_DIR"
 84fi
 85
 86cleanup() {
 87    if [[ "$KEEP_WORKTREE" -eq 1 ]]; then
 88        echo "Worktree preservado em: $WORKTREE_DIR"
 89        return
 90    fi
 91
 92    if [[ "$WORKTREE_ADDED" -eq 1 ]]; then
 93        git -C "$REPO_ROOT" worktree remove --force "$WORKTREE_DIR" >/dev/null 2>&1 || true
 94    fi
 95    rm -rf "$WORKTREE_DIR" >/dev/null 2>&1 || true
 96}
 97trap cleanup EXIT
 98
 99echo "Criando worktree isolado em: $WORKTREE_DIR (ref: $REF)"
100git -C "$REPO_ROOT" worktree add --detach "$WORKTREE_DIR" "$REF" >/dev/null
101WORKTREE_ADDED=1
102
103FLOW_DIR="$WORKTREE_DIR/.specs/codebase"
104if [[ ! -f "$FLOW_DIR/fluxo_sismo.sh" ]]; then
105    echo "Erro: '$FLOW_DIR/fluxo_sismo.sh' nao encontrado no worktree." >&2
106    exit 1
107fi
108
109echo "Executando em ambiente isolado:"
110echo "  cd $FLOW_DIR && bash ./fluxo_sismo.sh $*"
111
112set +e
113(
114    cd "$FLOW_DIR"
115    bash ./fluxo_sismo.sh "$@"
116)
117RC=$?
118set -e
119
120if [[ "$RC" -eq 0 ]]; then
121    echo "fluxo_sismo.sh concluiu com sucesso."
122else
123    echo "fluxo_sismo.sh finalizou com erro (exit code: $RC)." >&2
124fi
125
126exit "$RC"
