  1#!/usr/bin/env bash

  4usage() {
  5    cat <<'EOF'
  6Uso:
  7  scripts/dev/run_fluxo_test_matrix_archived.sh [opcoes]
  8
  9Executa a matriz padrao de testes do fluxo_sismo.sh em uma copia isolada
 10(/tmp) e arquiva os artefatos no repositorio para validacao historica.
 11
 12Opcoes:
 13  --catalog <arquivo>       Catalogo em arquivos/catalogo (default: catalogo_jul_filtrado.csv)
 14  --test-limit <n>          Limite de eventos no modo --test (default: 50)
 15  --core-env <nome>         PYENV_VERSION base do fluxo (default: sismo-core-311)
 16  --rnc-env <nome>          Ambiente da etapa -pr (default: sismo-rnc-379)
 17  --cases <lista>           Casos separados por virgula
 18                            opcoes: pre,eventos,predict,pos,maps,report,todos
 19                            default: pre,eventos,predict,pos,maps,report,todos
 20  --archive-root <path>     Diretorio de arquivamento (default: .specs/.../test_matrix)
 21  --archive-id <id>         Nome do subdiretorio de arquivamento (default: timestamp)
 22  --run-dir <path>          Diretorio isolado temporario (default: mktemp em /tmp)
 23  --keep-run-dir            Nao remove o diretorio isolado ao final
 24  --skip-results            Arquiva apenas logs/summary (nao copia CSV/PDF de resultados)
 25  --dry-run                 Mostra comandos sem executar
 26  -h, --help                Mostra esta ajuda
 27
 28Exemplos:
 29  scripts/dev/run_fluxo_test_matrix_archived.sh
 30  scripts/dev/run_fluxo_test_matrix_archived.sh --cases pre,predict,report --test-limit 25
 31  scripts/dev/run_fluxo_test_matrix_archived.sh --archive-id 20260210_155407
 32EOF
 33}

195run_case() {
196    local name="$1"
197    local timeout_s
198    local args=()
199    local log_file
200    local start
201    local rc=0
202    local duration
203
204    case "$name" in
205        pre) timeout_s=240; args=(-pe -t) ;;
206        eventos) timeout_s=120; args=(-e -t) ;;
207        predict) timeout_s=300; args=(-pr -t) ;;
208        pos) timeout_s=240; args=(-po -t) ;;
209        maps) timeout_s=180; args=(-m -t) ;;
210        report) timeout_s=240; args=(-r -t) ;;
211        todos) timeout_s=900; args=(-pe -e -pr -po -m -r -t) ;;
212        *) die "caso nao suportado: $name" ;;
213    esac
214
215    log_file="$RUN_DIR/arquivos/registros/test_matrix/${name}_${ARCHIVE_ID}.log"
216    start="$(date +%s)"
217
218    if [[ "$DRY_RUN" -eq 1 ]]; then
219        echo "[dry-run] env TEST_EVENT_LIMIT=$TEST_LIMIT PYENV_VERSION=$CORE_ENV RNC_PYENV_VERSION=$RNC_ENV timeout -k 10 $timeout_s bash ./fluxo_sismo.sh $CATALOG ${args[*]} > $log_file 2>&1"
220        rc=0
221    else
222        if env \
223            TEST_EVENT_LIMIT="$TEST_LIMIT" \
224            PYENV_VERSION="$CORE_ENV" \
225            RNC_PYENV_VERSION="$RNC_ENV" \
226            timeout -k 10 "$timeout_s" \
227            bash "$RUN_DIR/fluxo_sismo.sh" "$CATALOG" "${args[@]}" > "$log_file" 2>&1; then
228            rc=0
229        else
230            rc=$?
231        fi
232    fi
233
234    duration="$(( $(date +%s) - start ))"
235    if [[ "$DRY_RUN" -eq 1 ]]; then
236        echo "[dry-run] append summary line for $name"
237    else
238        printf '%s\t%s\t%s\t%s\t%s\n' "$name" "$rc" "$duration" "$timeout_s" "arquivos/registros/test_matrix/$(basename "$log_file")" >> "$SUMMARY_FILE"
239    fi
240    log "case=$name rc=$rc duration=${duration}s log=$(basename "$log_file")"
241}
242
243for case_name in "${CASES[@]}"; do
244    run_case "$case_name"
