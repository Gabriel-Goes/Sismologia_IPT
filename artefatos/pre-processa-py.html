 72def data_catalogo(
 73        catalogo: pd.DataFrame,
 74        ascending: bool = False,
 75        data: str = '2010'
 76) -> pd.DataFrame:
 77    catalogo['Time'] = pd.to_datetime(catalogo['Time'])
 78    catalogo['Hora'] = catalogo['Time'].apply(lambda x: x.hour)
 79    catalogo['Time'] = pd.to_datetime(catalogo['Time'])
 80    catalogo.sort_values(by='Time', ascending=ascending, inplace=True)
 81    catalogo = catalogo[catalogo['Time'] > data + '-01-01']
 82    print(catalogo['Time'].min())
 83
 84    return catalogo

 87def brasil_catalogo(catalogo: pd.DataFrame) -> pd.DataFrame:
 88    df = catalogo.copy()
 89    df['geometry'] = df.apply(
 90        lambda x: shapely.geometry.Point(x['Longitude'], x['Latitude']), axis=1
 91    )
 92    df = gpd.GeoDataFrame(df, geometry='geometry')
 93    world = gpd.read_file('arquivos/figuras/mapas/shp/ne_110m_admin_0_countries.shp')
 94    brasil = world[world['SOVEREIGNT'] == 'Brazil']
 95    brasil = brasil.to_crs(epsg=32723)
 96    brasil_buffer = brasil.buffer(400000)
 97    brasil_buffer = brasil_buffer.to_crs(epsg=4326)
 98    catalog_br = df[df.within(brasil_buffer.unary_union)]
 99
100    return catalog_br

103def filter_pred_com(pred: pd.DataFrame) -> pd.DataFrame:
104    pred = pred.reset_index()
105    pred['Hora'] = pred['Event'].apply(lambda x: UTCDateTime(x).hour)
106    pred_com = pred[(pred['Hora'] >= 11) & (pred['Hora'] < 22)]
107    pred_nc = pred[(pred['Hora'] < 11) | (pred['Hora'] >= 22)]
108    pred_com = pred_com.drop(columns=['Hora'])
109    pred_nc = pred_nc.drop(columns=['Hora'])
110    pred_com.to_csv('arquivos/resultados/test_comm.csv', index=False)
111    pred_nc.to_csv('arquivos/resultados/test_ncomm.csv', index=False)
112
113    return pred, pred_com, pred_nc

130def plot_distrib_hora(
131        catalog: pd.DataFrame, title='completo',
132        textwidth=7, scale=1.0, aspect_ratio=6 / 8
133) -> None:
134    print('')
135    print('Plotando distribuição de eventos por hora...')
136    data_i = catalog['Time'].min().strftime('%b de %Y')
137    data_f = catalog['Time'].max().strftime('%b de %Y')
138    counts = catalog['Hora'].value_counts(sort=False).reindex(np.arange(24), fill_value=0)
139    print(f' Catálogo {title} com {catalog.shape[0]} eventos')
140    print(f' {data_i} à {data_f}')
141    density = counts / counts.sum()
142    width = textwidth * scale
143    height = width * aspect_ratio
144    fig, ax = plt.subplots(figsize=(width, height), constrained_layout=True)
145    colors = ['#1f77b4' if (11 <= hour < 22) else '#ff7f0e' for hour in counts.index]
146    bars = ax.bar(
147        counts.index,
148        density,
149        color=colors, alpha=0.7, width=0.8, edgecolor='black'
150    )
151    yval_max = 0
152    for b, l in zip(bars, counts):
153        yval = b.get_height()
154        yval_max = yval if yval > yval_max else yval_max
155        ax.text(
156            b.get_x() + b.get_width() / 2,
157            yval + 0.002,
158            int(l),
159            ha='center', va='bottom', fontsize=8
160        )
161    plt.line = ax.axhline(
162        y=0.04166, color='red', linewidth=0.4, alpha=0.6, linestyle='--')
163    ax.legend(
164        [bars[0], bars[12]],
165        ['Horário Não Comercial', 'Horário Comercial'],
166        loc='best',
167        fontsize=8, fancybox=False,
168        edgecolor="black").get_frame().set_linewidth(0.5)
169    ax.yaxis.set_major_formatter(plt.FuncFormatter(lambda y, _: '{:.0%}'.format(y)))
170    ax.set_xticks(range(0, 24))
171    ax.set_xticklabels(range(0, 24), fontsize=8)
172    ax.set_yticklabels(
173        ['{:,.0%}'.format(x) for x in ax.get_yticks()],
174        fontsize=8
175    )
176    # ax.set_yticklabels(['{:,.0%}'.format(x) for x in ax.get_yticks()], fontsize=8)
177    ax.grid(axis='y', linestyle='--', linewidth=0.7)
178    fig.suptitle('Distribuição de Eventos por Hora', fontsize=10)
179    ax.set_title(f'Catálogo {title} com {catalog.shape[0]} eventos de {data_i} à {data_f}', fontsize=8)
180    ax.set_xlabel('Hora (UTC)', fontsize=9)
181    ax.set_ylabel('Frequência (%)', fontsize=9)
182    ax.set_ylim(0, 0.07)
183#     plt.savefig(f'arquivos/figuras/pre_processa/hist_hora_{title}.pgf', dpi=300)
184    plt.savefig(f'arquivos/figuras/pre_processa/hist_hora_{title}.png', dpi=300)
185    plt.close()

217def plot_out_of_brasil_as_red(catalog: pd.DataFrame) -> None:
218    catalog.drop_duplicates(subset='', inplace=True)
219    catalog = gpd.GeoDataFrame(
220        catalog,
221        geometry=gpd.points_from_xy(catalog.Longitude, catalog.Latitude)
222    )
223    catalog.crs = 'EPSG:4326'
224    brasil = gpd.read_file('arquivos/figuras/mapas/macrorregioesBrasil.json')
225    brasil.to_crs(epsg=4326, inplace=True)
226    brasil = brasil.geometry.unary_union
227    catalog['color'] = 'blue'
228    catalog.loc[~catalog.geometry.within(brasil), 'color'] = 'red'
229    fig, ax = plt.subplots()
230    catalog.plot(ax=ax, color=catalog['color'], markersize=10)
231    gpd.GeoSeries([brasil]).boundary.plot(ax=ax, color='black')
232    plt.title('Seismic Events: Red if outside Brazil, Blue if inside')
233    plt.xlabel('Longitude')
234    plt.ylabel('Latitude')
235    plt.grid(True)
236    plt.savefig('arquivos/figures/mapas/completo_catalog_mapa.png')
237    plt.close()

345def plot_by_macrorregioes(
346        catalog: pd.DataFrame,
347        title: str = 'completo',
348        attribute: str = 'Depth/km',
349        textwidth=7,
350        scale=1.0,
351        aspect_ratio=6 / 8,
352        bins=10
353) -> None:
354    width = textwidth * scale
355    height = width * aspect_ratio
356    macro_br = gpd.read_file('arquivos/figuras/mapas/macrorregioesBrasil.json')
357    macro_br = macro_br.to_crs(epsg=4326)
358    regions = list(macro_br['nome']) + ['Exterior']
359    num_regions = len(regions)
360    cols = 3
361    rows = (num_regions + cols - 1) // cols
362    fig, axes = plt.subplots(
363        rows, cols,
364        figsize=(width, height),
365        constrained_layout=True
366    )
367    axes = axes.flatten()
368    all_geometries = macro_br['geometry'].unary_union
369    for i, regiao in enumerate(regions):
370        ax = axes[i]
371        ax.set_title(regiao, fontsize=10)
372        ax.set_yscale('log')
373        ax.grid(True, linestyle='--', linewidth=0.7, alpha=0.7)
374        ax.set_xlabel('Profundidade (km)', fontsize=8)
375        ax.set_ylabel('Frequência', fontsize=8)
376        if regiao != 'Exterior':
377            data = catalog[attribute][catalog['geometry'].within(
378                macro_br.loc[macro_br['nome'] == regiao, 'geometry'].values[0]
379            )]
380        else:
381            data = catalog[attribute][~catalog['geometry'].within(
382                all_geometries
383            )]
384        counts, bin_edges, _ = ax.hist(
385            data,
386            bins=bins,
387            color='#1f77b4',
388            edgecolor='black',
389            alpha=0.7,
390            density=False
391        )
392        for j in range(len(bin_edges) - 1):
393            bin_value = counts[j] / catalog.shape[0] * 100
394            ax.text(
395                (bin_edges[j] + bin_edges[j + 1]) / 2,
396                counts[j],
397                f'{bin_value:.2f}%',
398                ha='center',
399                va='bottom',
400                fontsize=6,
401            )
402    for ax in axes[num_regions:]:
403        fig.delaxes(ax)
404    fig.suptitle('Distribuição de Profundidade por Macrorregiões', fontsize=12)
405    plt.subplots_adjust(top=0.92)
406    # plt.show()

410def main(args=args):
411    print(f'Argumentos recebidos:\n {args.eventos}\n {args.csv}\n {args.map}\n {args.test}')
412    if args.csv:
413        catalogo_r = read_catalogo(args.csv, sep='|')
414        if args.test is True:
415            print('Testando...')
416            random.seed(42)
417            catalogo_r = catalogo_r.sample(catalogo_r.shape[0] // 3)
418        catalogo = data_catalogo(catalogo_r)
419        catalogo = brasil_catalogo(catalogo)
420        catalogo['Author'] = catalogo['Author'].str.upper()
421        catalogo['Author'] = catalogo['Author'].str.strip()
422        catalogo['Author'] = catalogo['Author'].replace(
423            {
424                'BRUNO@LAB88': 'BRUNO',
425                'BBCOLLACO': 'BRUNO',
426                'BRUNO@MAVERICK.LOCAL': 'BRUNO',
427                'MARCELO': 'MASSUMPCAO',
428                'JROBERTO@VITORIA': 'JROBERTO',
429            }
430        )
431        plot_distrib_hora(catalogo, title='filtrado')
432        plot_distrib_hora(catalogo_r)
433        profundidade_catalogo(catalogo_r)
434        profundidade_catalogo(catalogo, title='filtrado')
435        c_filtrado = f"arquivos/catalogo/{args.csv.split('.')[0]}_filtrado.csv"
436        catalogo.to_csv(
437            c_filtrado,
438            sep='|',
439            index=False
440        )
441        if args.map:
442            print('Plotando mapa...')
443            plot_prof_as_red(catalogo_r)
444            plot_prof_as_red(catalogo, 'filtrado')
445
446    elif args.eventos:
447        eventos = pd.read_csv(f'arquivos/eventos/{args.eventos}', sep=',')
448        if args.map:
449            plot_prof_as_red(eventos)
450
451    else:
452        print('Nenhum argumento foi passado')
453        return None
454
455    plt.close()
456    return catalogo
