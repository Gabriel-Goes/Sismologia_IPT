 22def plot_pred_map(data, filename):
 23    # Definir região do mapa
 24    region = [
 25        data['Longitude'].min() - 1, data['Longitude'].max() + 1,
 26        data['Latitude'].min() - 1, data['Latitude'].max() + 1
 27    ]
 28    fig = pygmt.Figure()
 29    pygmt.makecpt(
 30        cmap="polar", series=[data['Event Prob_Nat'].min(), data['Event Prob_Nat'].max()]
 31    )
 32    fig.basemap(region=region, projection='M15c', frame=True)
 33    fig.coast(land="lightgray", water="skyblue")
 34
 35    # Adicionar jitter aos dados para evitar sobreposição
 36    jitter_strength = 0.1  # Ajuste conforme necessário
 37    data['Longitude_jitter'] = data['Longitude'] + np.random.uniform(-jitter_strength, jitter_strength, size=len(data))
 38    data['Latitude_jitter'] = data['Latitude'] + np.random.uniform(-jitter_strength, jitter_strength, size=len(data))
 39
 40    fig.plot(
 41        x=data['Longitude_jitter'],
 42        y=data['Latitude_jitter'],
 43        fill=data['Event Prob_Nat'],
 44        style='c0.1c',  # Diminuir o tamanho dos pontos
 45        cmap=True,
 46        transparency=50  # Adicionar transparência
 47    )
 48
 49    # Adicionar título ao mapa
 50    fig.text(
 51        x=0.5 * (region[0] + region[1]),
 52        y=region[3] - 0.5,
 53        text=f"Mapa de Predições de {data.shape[0]} Eventos rotulados como Naturais",
 54        font="12p,Helvetica-Bold",
 55        justify="CM",
 56    )
 57
 58    # Adicionar barra de cores
 59    fig.colorbar(frame=["x+lProbabilidade", "y+lm"])
 60
 61    # Adicionar seta de norte
 62    fig.basemap(rose="jTL+w2c+o0.5c/0.5c+f")
 63
 64    # Salvar e mostrar figura
 65    fig.savefig(f'arquivos/figuras/mapas/{filename}')
 66    fig.show()

 69def plot_macroregions(gdf, data):
 70    gdf = gdf.to_crs(epsg=4326)
 71    jitter_strength = 0.01  # Ajuste conforme necessário
 72    data['Longitude_jitter'] = data['Longitude'] + np.random.uniform(-jitter_strength, jitter_strength, size=len(data))
 73    data['Latitude_jitter'] = data['Latitude'] + np.random.uniform(-jitter_strength, jitter_strength, size=len(data))
 74
 75    for idx, row in gdf.iterrows():
 76        nome_macroregiao = row["nome"]
 77        fig = pygmt.Figure()
 78        bounds = row.geometry.bounds
 79        lon_min, lat_min, lon_max, lat_max = bounds
 80        if lon_max - lon_min > 360 or lat_max - lat_min > 180:
 81            raise ValueError("Os limites da região excedem os valores válidos")
 82
 83        region = [
 84            max(-180, lon_min - 1), min(180, lon_max + 1),
 85            max(-90, lat_min - 1), min(90, lat_max + 1)
 86        ]
 87        fig.basemap(region=region, projection="M6i", frame=True)
 88        fig.coast(
 89            shorelines=True,
 90            resolution='10m',
 91            borders=[1],
 92            water='skyblue'
 93        )
 94        fig.plot(
 95            x=data['Longitude_jitter'],
 96            y=data['Latitude_jitter'],
 97            style='c0.1c',
 98            fill='red',
 99            transparency=50  # Adicionar transparência
100        )
101
102        # Adicionar título ao mapa
103        fig.text(
104            x=0.5 * (region[0] + region[1]),
105            y=region[3] - 0.5,
106            text=f"Mapa da Macrorregião {nome_macroregiao}",
107            font="12p,Helvetica-Bold",
108            justify="CM",
109        )
110
111        # Adicionar seta de norte
112        fig.basemap(rose="jTL+w2c+o0.5c/0.5c+f")
113
114        # Adicionar legenda
115        with tempfile.NamedTemporaryFile(mode="w+", delete=False) as temp_legend_file:
116            legend_spec = f"""
117H 12p,Helvetica Legenda
118D 0.2c 1p
119N 2
120S 0.3c c 0.15c red 0.1p 0.5c Eventos
121"""
122            temp_legend_file.write(legend_spec)
123            temp_legend_file_name = temp_legend_file.name
124
125        legend_width = "10c"
126        position = "JBC+o10c/1c+w" + legend_width
127        fig.legend(position=position, spec=temp_legend_file_name, box="+gwhite+p1p")
128
129        fig.savefig(f"arquivos/figuras/mapas/{nome_macroregiao}_pred.png")
130        fig.show()
131        os.remove(temp_legend_file_name)

134def main():
135    data = pd.read_csv("arquivos/resultados/nc_analisado_final.csv")
136    data = data.groupby(['Event']).first()
137    plot_pred_map(data, "mapa.png")
138
139    gdf = gpd.read_file("arquivos/figuras/mapas/macrorregioesBrasil.json")
140    plot_macroregions(gdf, data)

142    return data, gdf
