  1#!/bin/bash
  2# -*- coding: utf-8 -*-
  3#
  4# -----------------------------------------------------------------------------
  5# ./ClassificadorSismologico/fluxo_sismo.sh
  6# Autor: Gabriel Góes Rocha de Lima

 11
 12# ----------------------------  DESCRIÇÃO  -----------------------------------
 13# Este código recebe uma lista de IDs eventos ou uma data de início e fim e com
 14# estas informações cria um arquivo eventos.csv com os eventos sísmicos que
 15# ocorreram no intervalo de tempo especificado ou com os IDs presentes na lista
 16# passada e foram capturados pela rede sismológica. Com este arquivo, segue-se
 17# para o passo de aquisição das formas de onda e classificação.
 18
 19# -------------------------------  USO -----------------------------------------
 20# Passando uma lista de IDs de eventos sísmicos:
 21#       Sismo_Pipeline.sh catalogo.csv
 22#
 23# Caso utilize uma lista de IDs, o script irá adquirir os eventos sísmicos da
 24# rede seisarc e criará um arquivo eventos.csv
 25
 26# -----------------------------  VARIÁVEIS ------------------------------------
 27EVENTS=False
 28TREATCATALOG=False
 29PREDICT=False
 30POSPROCESS=False
 31MAPS=False
 32REPORT=False
 33TEST=False
 34RNC_PYENV_VERSION=${RNC_PYENV_VERSION:-sismo-rnc-379}
 35
 36# Resolve diretório base automaticamente para o baseline movido em .specs/codebase.
 37SCRIPT_DIR="$(cd "$(dirname "${BASH_SOURCE[0]}")" && pwd)"
 38BASE_DIR=${BASE_DIR:-"$SCRIPT_DIR"}
 39
 40# ------------------------- PARSE ARGUMENTOS ----------------------------------
 41if [ $# -eq 0 ]; then # Não está funcionando.
 42    ## Deve configurar tudo como true caso não seja passado argumentos ao ./flusho_sismo.sh.
 43    ## Porém, para rodar o bash, é necessário passar o nome do arquivo de
 44    ## catalogo_MÊS.csv.
 45    EVENTS=True
 46    TREATCATALOG=True
 47    PREDICT=True
 48    POSPROCESS=True
 49    MAPS=True
 50    REPORT=True
 51    TEST=False
 52else
 53    if [ -f "$BASE_DIR/arquivos/catalogo/$1" ]; then
 54        CATALOG=$(basename "$1")
 55        echo "Catalogo: $CATALOG"
 56        shift
 57    else
 58        echo "Erro: Arquivo $1 não encontrado!"
 59        exit 1
 60    fi
 61    while [[ $# -gt 0 ]]; do
 62        case $1 in
 63            --help|-h)
 64                echo "Uso: $0 [opções] [catalogo.csv]"
 65                echo "Opções:"
 66                echo "  --eventos, -e:    Executa a aquisição de eventos sísmicos"
 67                echo "  --pre, -pe:       Executa o tratamento do catálogo"
 68                echo "  --predict, -pr:   Executa o script de predição"
 69                echo "  --pos, -po:       Executa o pós-processamento"
 70                echo "  --maps, -m:       Executa a geração de mapas"
 71                echo "  --test, -t:       Executa um bach de testes com 500 eventos"
 72                echo "  --report, -r:     Executa a geração de relatórios"
 73                exit 0
 74                ;;
 75            --eventos|-e)
 76                EVENTS=True
 77                ;;
 78            --pre|-pe)
 79                TREATCATALOG=True
 80                ;;
 81            --predict|-pr)
 82                PREDICT=True
 83                ;;
 84            --pos|-po)
 85                POSPROCESS=True
 86                ;;
 87            --maps|-m)
 88                MAPS=True
 89                ;;
 90            --test|-t)
 91                TEST=True
 92                ;;
 93            --report|-r)
 94                REPORT=True
 95                ;;
 96            *)
 97                echo "Opção inválida: $1"
 98                exit 1
 99                ;;
100        esac
101        shift
102    done
103fi
104
105# -----------------------------  VARIÁVEIS -------------------------------------
106EVENTS=${EVENTS:-False}
107TREATCATALOG=${TREATCATALOG:-True}
108PREDICT=${PREDICT:-False}
109POSPROCESS=${POSPROCESS:-False}
110MAPS=${MAPS:-False}
111REPORT=${REPORT:-False}
112
113# ----------------------------- CONSTANTES -------------------------------------
114set -e
115# DEFINE OS DIRETÓRIOS DE TRABALHO
116pushd "$BASE_DIR"
117
118# Garante imports locais dos modulos em fonte/*.
119export PYTHONPATH="$BASE_DIR/fonte${PYTHONPATH:+:$PYTHONPATH}"
120
121# Em ambientes restritos, ~/.gmt pode nao ser gravavel para sessoes.
122# Mantemos userdir em /tmp e reaproveitamos a base local de geografia quando existir.
123export GMT_USERDIR="${GMT_USERDIR:-/tmp/gmt-${USER:-user}}"
124mkdir -p "$GMT_USERDIR"
125mkdir -p "$GMT_USERDIR/geography"
126
127# Em alguns hosts/sandboxes, GMT_USERDIR pode conter pastas vazias de tentativas anteriores.
128# Garantimos links para datasets locais de geografia quando disponiveis no HOME.
129for _dataset in gshhg dcw; do
130    _src="${HOME}/.gmt/geography/${_dataset}"
131    _dst="${GMT_USERDIR}/geography/${_dataset}"
132    if [ -d "$_src" ]; then
133        if [ "$_dataset" = "gshhg" ] && [ -f "$_dst/binned_GSHHS_i.nc" ]; then
134            continue
135        fi
136        if [ ! -L "$_dst" ]; then
137            rm -rf "$_dst" 2>/dev/null || true
138            ln -s "$_src" "$_dst" 2>/dev/null || cp -a "$_src" "$_dst"
139        fi
140    fi
141done
142
143# DEFINE O DIRETÓRIO DE LOGS
144LOG_DIR="arquivos/registros"
145LOG_FILE="$LOG_DIR/Sismo_Pipeline.log"
146mkdir -p "$LOG_DIR/.bkp"
147
148if [ -f "$LOG_FILE" ]; then
149    mv "$LOG_FILE" "$LOG_DIR/.bkp/Sismo_Pipeline.log.$(date +%Y%m%d%H%M%S)"
150    touch "$LOG_FILE"
151fi
152
153exec 1> >(tee -a "$LOG_FILE") 2>&1
154
155# DEFINE DELIMITADORES PARA LOGS
156DELIMT1='########################################################################'
157DELIMT2='========================================================================'
158
159# ---------------------------- INICIO DO PIPELINE  ------------------------------
160echo ''
161echo $DELIMT2
162echo "                       Iniciando do Pipeline                         "
163echo $DELIMT2
164echo ''
165
166# PRINT ARGUMENTS RECEIVED AND DEFAULTS
167echo "Argumentos recebidos:"
168echo "  Catalogo: $CATALOG"
169echo "  EVENTS: $EVENTS"
170echo "  TREATCATALOG: $TREATCATALOG"            # Código para ajuda
171echo "  PREDICT: $PREDICT"
172echo "  POSPROCESS: $POSPROCESS"
173echo "  MAPS: $MAPS"
174echo "  REPORT: $REPORT"
175echo "  TEST: $TEST"
176echo ''
177
178# ------------------------- ETAPA DE AQUISIÇÃO DE DADOS  ----------------------
179if [ "$TREATCATALOG" = True ]; then
180        echo ''
181        echo " -------------- INICIANDO O TRATAMENTO -------------------- "
182        echo " Tratando $CATALOG..."
183        echo " t0: $(date)"
184        python fonte/analise_dados/pre_processa.py -c $CATALOG
185fi
186
187if [ "$EVENTS" = True ]; then
188    echo " Criando arquivos de backup..."
189    mkdir -p arquivos/eventos/.bkp arquivos/registros/.bkp
190    [[ -f arquivos/eventos/eventos.csv ]] &&
191        cp arquivos/eventos/eventos.csv arquivos/eventos/.bkp/eventos.csv.$(date +%Y%m%d%H%M%S)
192    [[ -f arquivos/registros/ids_faltantes.csv ]] &&
193        cp arquivos/registros/ids_faltantes.csv arquivos/registros/.bkp/ids_faltantes.csv.$(date +%Y%m%d%H%M%S)
194    echo " Arquivos de backup criados com sucesso!"
195    echo ''
196    echo ' -> Executando fluxo_eventos.py...'
197    echo $CATALOG
198    python fonte/nucleo/fluxo_eventos.py $CATALOG $TEST
199    echo ''
200fi
201
202# ------------------------- ETAPA DE PREDIÇÃO  ----------------------------------
203if [ "$PREDICT" = True ]; then
204    echo " ----------------- INICIANDO O PREDICT.PY ---------------------------- "
205    echo " Ambiente RNC (somente predição): PYENV_VERSION=$RNC_PYENV_VERSION"
206    if [ "$TEST" = True ]; then
207        TEST_EVENT_LIMIT=${TEST_EVENT_LIMIT:-50}
208        echo " Modo de teste RNC: limite de $TEST_EVENT_LIMIT eventos"
209        env PYENV_VERSION="$RNC_PYENV_VERSION" python fonte/rnc/run.py --test-limit "$TEST_EVENT_LIMIT"
210    else
211        env PYENV_VERSION="$RNC_PYENV_VERSION" python fonte/rnc/run.py
212    fi
213    echo ''
214fi
215
216# --------- ETAPA DE GERAR GRAFICOS E ANÁLISES -----------
217if [ "$POSPROCESS" = True ]; then
218    echo " ---------------- INICIANDO DATA_ANALYSIS/POSPROCESS.PY ---------------------------- "
219    python fonte/analise_dados/pos_processa.py
220    echo ''
221fi
222
223# ------------------------- ETAPA DE GERAR MAPAS  -----------------------------
224# Condicionalmente executa partes do script
225if [ "$MAPS" = True ]; then
226    echo " -------------- Processo de criação de mapas iniciado ------------------"
227    # Checa se o arquivo de eventos existe e se é vazio
228    if [ -f "arquivos/output/no_commercial/df_nc_pos.csv" ]; then
229        echo " -> Executando make_maps.py..."
230        python fonte/analise_dados/gera_mapas.py
231    fi
232    echo ''
233fi
234
235# ----------------- ETAPA DE GERAR RELATORIOS ------------------------
236if [ "$REPORT" = True ]; then
237    echo " ----------------- Iniciando o pdflatex .tex ---------------------------- "
238    python fonte/relatorio-sismologia/pyscripts/figures.py --path 'pre_processa'
239    python fonte/relatorio-sismologia/pyscripts/figures.py --path 'pos_processa'
240    python fonte/relatorio-sismologia/pyscripts/mapa.py
241    pushd fonte/relatorio-sismologia
242    RESULT_DIR=${RESULT_DIR:-"$BASE_DIR/arquivos/resultados/relatorios"}
243    pdflatex -interaction=nonstopmode -halt-on-error \
244        -output-directory="$RESULT_DIR" relatorio_preditivo.tex
245    popd
246fi
247
248echo $DELIMT2
249echo "                        Fim do Pipeline                                 "
250echo $DELIMT2
251echo ''
252popd
