 500def calc_event_prob(df, z_values):
 501    df['Event Prob_Nat_f'] = [{} for _ in range(len(df))]
 502    for z in z_values:
 503        for ev_index, evento in df.groupby('Event').groups.items():
 504            prob_nat_x = 0
 505            counter = 0
 506            evento_df = df.loc[evento]
 507            for _, pick in evento_df.groupby('Station'):
 508                if pick['CFT'].values[0] < z:
 509                    continue

1286def snr(eventos: pd.DataFrame,
1287        window: int) -> [pd.DataFrame, dict]:
1288    eventos.sort_index(inplace=True)
1289    filtro = AttribDict({
1290        'pa': 2.0, 'pb': 49.0,
1291        'noise': '', 'p': '', 's': '', 'snrp': '', 'snrs': ''
1292    })
1293    for index, pick in tqdm(eventos.iterrows()):
1294        nw = str(obspy.UTCDateTime(pick['Pick Time']) - 4.9) + '/4'
1295        pw = str(obspy.UTCDateTime(pick['Pick Time'])) + '/' + str(window)
1296        sw = str(obspy.UTCDateTime(pick['Pick Time']) + 20) + '/5'
1297        noisewindow = parsewindow(nw)
1298        pwindow = parsewindow(pw)
1299        swindow = parsewindow(sw)
1300        st = obspy.read(f"arquivos/mseed/{pick['Path']}")
1301        noise, trace_p, trace_s = prepare(
1302            st[2],
1303            filtro,
1304            noisewindow, pwindow, swindow
1305        )
1306        filtro.noise = np.mean(np.abs(noise))
1307        filtro.p = np.mean(np.abs(trace_p))
1308        filtro.s = np.mean(np.abs(trace_s))
1309        if filtro.noise == 0:
1310            filtro.snrp = np.nan
1311            filtro.snrs = np.nan
1312            filtro.noise = np.nan
1313            eventos.loc[index, 'Error'] = 'Noise = 0'
1314            continue
1315        filtro.snrp = filtro.p / filtro.noise
1316        filtro.snrs = filtro.s / filtro.noise
1317        eventos.loc[index, 'SNR_P'] = filtro.snrp
1318        eventos.loc[index, 'SNR_S'] = filtro.snrs
1319        eventos.loc[index, 'Noise'] = filtro.noise
1320        eventos.loc[index, 'p'] = filtro.p
1321
1322    return eventos
