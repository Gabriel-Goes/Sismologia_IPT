  1#!/usr/bin/env bash
  2set -euo pipefail
  3
  4usage() {
  5    cat <<'EOF'
  6Uso:
  7  scripts/dev/import_test_matrix_to_docs.sh [opcoes]
  8
  9Importa artefatos de uma execucao em
 10.specs/codebase/arquivos/registros/test_matrix/<archive_id>
 11para docs/operacoes/anexos e cria/atualiza paginas em docs/sphinx/source/anexos.
 12
 13Opcoes:
 14  --archive-id <id>      ID da execucao (default: ultimo diretorio em test_matrix)
 15  --op-tag <tag>         Prefixo dos anexos (ex.: 0025b, 0026a) [obrigatorio]
 16  --files <lista>        Itens separados por virgula
 17                         default: summary,manifest,predict,pos,report,pre_processado,predito,analisado_final,erros
 18  --dry-run              Mostra acoes sem alterar arquivos
 19  -h, --help             Mostra ajuda
 20
 21Itens suportados em --files:
 22  summary,manifest,pre,eventos,predict,pos,maps,report,todos,
 23  pre_processado,predito,analisado_final,erros
 24EOF
 25}
 26
 27die() {
 28    echo "[import-test-matrix] ERRO: $*" >&2
 29    exit 1
 30}
 31
 32log() {
 33    echo "[import-test-matrix] $*"
 34}
 35
 36run_cmd() {
 37    if [[ "$DRY_RUN" -eq 1 ]]; then
 38        echo "[dry-run] $*"
 39        return 0
 40    fi
 41    "$@"
 42}

 44to_hyphen() {
 45    echo "$1" | tr '_' '-'
 46}

 48SCRIPT_DIR="$(cd "$(dirname "${BASH_SOURCE[0]}")" && pwd)"
 49REPO_ROOT="$(git -C "$SCRIPT_DIR" rev-parse --show-toplevel 2>/dev/null || true)"
 50[[ -n "$REPO_ROOT" ]] || die "nao foi possivel localizar a raiz do repositorio."
 51
 52ARCHIVE_ROOT="$REPO_ROOT/.specs/codebase/arquivos/registros/test_matrix"
 53DOCS_ANEXOS="$REPO_ROOT/docs/operacoes/anexos"
 54SPHINX_ANEXOS="$REPO_ROOT/docs/sphinx/source/anexos"
 55SPHINX_INDEX="$SPHINX_ANEXOS/index.rst"
 56
 57ARCHIVE_ID=""
 58OP_TAG=""
 59FILES_CSV="summary,manifest,predict,pos,report,pre_processado,predito,analisado_final,erros"
 60DRY_RUN=0
 61
 62while [[ $# -gt 0 ]]; do
 63    case "$1" in
 64        --archive-id)
 65            ARCHIVE_ID="${2:-}"
 66            [[ -n "$ARCHIVE_ID" ]] || die "--archive-id requer valor."
 67            shift 2
 68            ;;
 69        --op-tag)
 70            OP_TAG="${2:-}"
 71            [[ -n "$OP_TAG" ]] || die "--op-tag requer valor."
 72            shift 2
 73            ;;
 74        --files)
 75            FILES_CSV="${2:-}"
 76            [[ -n "$FILES_CSV" ]] || die "--files requer valor."
 77            shift 2
 78            ;;
 79        --dry-run)
 80            DRY_RUN=1
 81            shift
 82            ;;
 83        -h|--help)
 84            usage
 85            exit 0
 86            ;;
 87        *)
 88            die "opcao invalida: $1"
 89            ;;
 90    esac
 91done
 92
 93[[ -n "$OP_TAG" ]] || die "--op-tag e obrigatorio."
 94[[ -d "$ARCHIVE_ROOT" ]] || die "diretorio nao encontrado: $ARCHIVE_ROOT"
 95[[ -f "$SPHINX_INDEX" ]] || die "indice Sphinx nao encontrado: $SPHINX_INDEX"
 96
 97if [[ -z "$ARCHIVE_ID" ]]; then
 98    ARCHIVE_ID="$(
 99        find "$ARCHIVE_ROOT" -mindepth 1 -maxdepth 1 -type d -printf '%f\n' \
100        | sort | tail -n 1
101    )"
102    [[ -n "$ARCHIVE_ID" ]] || die "nenhuma execucao encontrada em $ARCHIVE_ROOT"
103fi
104
105ARCHIVE_DIR="$ARCHIVE_ROOT/$ARCHIVE_ID"
106[[ -d "$ARCHIVE_DIR" ]] || die "execucao nao encontrada: $ARCHIVE_DIR"
107
108run_cmd mkdir -p "$DOCS_ANEXOS" "$SPHINX_ANEXOS"
109
110IFS=',' read -r -a ITEMS <<< "$FILES_CSV"
111[[ "${#ITEMS[@]}" -gt 0 ]] || die "nenhum item informado em --files."
112
113created=0
114skipped=0
115
116for item in "${ITEMS[@]}"; do
117    src=""
118    ext=""
119    case "$item" in
120        summary)
121            src="$ARCHIVE_DIR/summary_${ARCHIVE_ID}.tsv"; ext="tsv"
122            ;;
123        manifest)
124            src="$ARCHIVE_DIR/manifest.env"; ext="env"
125            ;;
126        pre|eventos|predict|pos|maps|report|todos)
127            src="$ARCHIVE_DIR/${item}_${ARCHIVE_ID}.log"; ext="log"
128            ;;
129        pre_processado)
130            src="$ARCHIVE_DIR/resultados/pre_processado.csv"; ext="csv"
131            ;;
132        predito)
133            src="$ARCHIVE_DIR/resultados/predito.csv"; ext="csv"
134            ;;
135        analisado_final)
136            src="$ARCHIVE_DIR/resultados/analisado_final.csv"; ext="csv"
137            ;;
138        erros)
139            src="$ARCHIVE_DIR/resultados/erros.csv"; ext="csv"
140            ;;
141        *)
142            die "item nao suportado em --files: $item"
143            ;;
144    esac
145
146    if [[ ! -f "$src" ]]; then
147        log "arquivo ausente para item '$item' (skip): $src"
148        skipped=$((skipped + 1))
149        continue
150    fi
151
152    dst_base="${OP_TAG}-${item}.${ext}"
153    dst_file="$DOCS_ANEXOS/$dst_base"
154    rst_stub="anexo-${OP_TAG}-$(to_hyphen "$item")-${ext}"
155    rst_file="$SPHINX_ANEXOS/${rst_stub}.rst"
156
157    run_cmd cp -f "$src" "$dst_file"
158
159    if [[ "$DRY_RUN" -eq 1 ]]; then
160        echo "[dry-run] gerar $rst_file"
161    else
162        cat > "$rst_file" <<EOF
163Anexo: ${dst_base}
164$(printf '%*s' $((7 + ${#dst_base})) '' | tr ' ' '=')
165
166Arquivo de origem:
167
168- :ghblob:\`docs/operacoes/anexos/${dst_base}\`
169
170Conteudo
171--------
172
173.. literalinclude:: ../../../../docs/operacoes/anexos/${dst_base}
174   :language: text
175   :linenos:
176EOF
177    fi
178
179    if [[ "$DRY_RUN" -eq 1 ]]; then
180        echo "[dry-run] garantir entrada no index: $rst_stub"
181    else
182        if ! grep -Fqx "   ${rst_stub}" "$SPHINX_INDEX"; then
183            echo "   ${rst_stub}" >> "$SPHINX_INDEX"
184        fi
185    fi
186
187    created=$((created + 1))
188    log "importado: $item -> $dst_base"
189done
190
191log "archive_id: $ARCHIVE_ID"
192log "op_tag    : $OP_TAG"
193log "itens ok  : $created"
194log "itens skip: $skipped"
