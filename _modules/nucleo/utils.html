def csv2list(csv_file: str, data=False) -> list:
    if data:
        evids = []
        with open(f'arquivos/catalogo/{csv_file}', 'r') as f:
            lines = f.readlines()
            evids_ = [line.split('|')[0] for line in lines[1:]]
        for evid in evids_:
            if int(evid[3:7]) > int(data):
                evids.append(evid)
        return evids

    else:
        with open(f'arquivos/catalogo/{csv_file}', 'r') as f:
            lines = f.readlines()
            return [line.split('|')[0] for line in lines[1:]]
